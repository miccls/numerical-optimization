% Special command for partial derivatives

% \d[order][function][variables]
\NewDocumentCommand{\pd}{O{} O{} O{}}{%
  \ensuremath{%
    \IfNoValueTF{#1}{% -------- no order
      \IfNoValueTF{#2}{%
        \partial % just \d
      }{%
        \IfNoValueTF{#3}{%
          \partial #2 % \d[f]
        }{%
          \frac{\partial #2}{\partial #3} % \d[f][x]
        }%
      }%
    }{% -------- order given
      \IfNoValueTF{#2}{%
        \partial^{#1} % \d[2]
      }{%
        \IfNoValueTF{#3}{%
          \partial^{#1} #2 % \d[2][f]
        }{%
          % build denominator for mixed partials
          \def\varlist{}%
          \renewcommand{\do}[1]{%
            \ifx\varlist\empty
              \def\varlist{\partial ##1}%
            \else
              \edef\varlist{\varlist \partial ##1}%
            \fi
          }%
          \docsvlist{#3}%
          \frac{\partial^{#1} #2}{\varlist} % e.g. \d[2][f][x,y]
        }%
      }%
    }%
  }% end ensuremath
}
